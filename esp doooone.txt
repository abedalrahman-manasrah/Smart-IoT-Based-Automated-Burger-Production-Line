void performMaintenance() {
  int activeCount = 0; int newCurrentIdx = -1;
  for (int i = 0; i < orderCount; i++) {
    if (!orders[i].completed) { orders[activeCount] = orders[i]; if (currentOrderIdx == i) { newCurrentIdx = activeCount; } activeCount++; }
  }
  orderCount = activeCount; currentOrderIdx = newCurrentIdx;
}

void processQueue() {
  if (isMachineWorking || isLocalOrderActive || isMaintenanceMode) return; 
  if (millis() - lastReadyTime < COOLDOWN_TIME) return; 

  for (int i = 0; i < orderCount; i++) {
    if (!orders[i].completed) {
      
      // 🌟 التعديل الجديد: فحص المخزون قبل إرسال الطلب للميجا وتحديد النقص 🌟
      String missing = "";
      if (mSt <= 0) missing = "اللحم";
      else if (bS_L <= 0) missing = "الخبز السفلي";
      else if (bS_H <= 0) missing = "الخبز العلوي";
      else if (orders[i].bits.charAt(0) == '1' && chS <= 0) missing = "الجبنة";
      else if (orders[i].bits.charAt(1) == '1' && oS <= 0) missing = "البصل";
      else if (orders[i].bits.charAt(2) == '1' && lS <= 0) missing = "الخس";
      else if (orders[i].bits.charAt(3) == '1' && tS <= 0) missing = "الطماطم";
      else if (orders[i].bits.charAt(4) == '1' && kSt <= 0) missing = "الكاتشب";
      else if (orders[i].bits.charAt(5) == '1' && mSt_sauce <= 0) missing = "المايونيز";

      // إذا كان هناك نقص، يتم إلغاء الطلب وكتابة اسم المكون الناقص
      if (missing != "") {
        orders[i].completed = true; 
        orders[i].progress = -2; // إشارة برمجية (كود) تفيد بأن الإلغاء بسبب نفاذ المخزون
        orders[i].type = missing; // استخدام خانة type لحفظ اسم المكون مؤقتاً وعرضه للزبون
        continue; // تخطي إرسال هذا الطلب والانتقال للطلب التالي في الطابور
      }
      // -----------------------------------------------------------------

      currentOrderIdx = i;
      isMachineWorking = true; 
      orders[i].progress = 5;
      
      orderStartTime = millis();

      String cmd = "<ORDER:" + orders[i].type + "|NAME:" + orders[i].name + "|BITS:" + orders[i].bits + ">";
      MegaSerial.println(cmd);
      return; 
    }
  }
  currentOrderIdx = -1; 
}

void loop() {
  dnsServer.processNextRequest(); 
  server.handleClient(); 
  
  yield(); 
  processQueue(); 

  if (millis() - lastMaintenanceTime > MAINTENANCE_INTERVAL) { performMaintenance(); lastMaintenanceTime = millis(); }

  // 🌟 إصلاح التايم أوت: تم رفع الوقت إلى 5 دقائق (300000 ملي ثانية) لتفادي الإلغاء المفاجئ للطلبات الكبيرة
  if (isMachineWorking && (millis() - orderStartTime > 300000)) { 
    isMachineWorking = false; isLocalOrderActive = false;
    if (currentOrderIdx != -1) { 
      orders[currentOrderIdx].progress = -1; 
      orders[currentOrderIdx].completed = true; 
    } 
    lastReadyTime = millis(); 
  }

  while (MegaSerial.available()) {
    String d = MegaSerial.readStringUntil('\n'); d.trim(); if (d.length() == 0) continue;
    
    if (d.startsWith("<MSG:")) { if (currentOrderIdx != -1 && !isLocalOrderActive) { orders[currentOrderIdx].progress = d.substring(5, d.indexOf(">")).toInt(); } } 
    else if (d.startsWith("<BUSY>")) { if (!isMachineWorking) { isLocalOrderActive = true; localQtyRemaining = 1; } }
    else if (d.startsWith("<KEYPAD_START:")) { int qty = d.substring(14, d.indexOf(">")).toInt(); localQtyRemaining = qty; isLocalOrderActive = true; }
    else if (d.startsWith("<READY>")) {
      // 🌟 التعديل هنا: زيادة العداد مع كل وجبة تكتمل سواء من الشاشة أو التطبيق
      totalOrdersCount++; 
      
      if (isLocalOrderActive) { localQtyRemaining--; if (localQtyRemaining <= 0) { isLocalOrderActive = false; } } 
      else if (currentOrderIdx != -1) {
        orders[currentOrderIdx].qtyRemaining--; 
        if (orders[currentOrderIdx].qtyRemaining <= 0) { orders[currentOrderIdx].completed = true; orders[currentOrderIdx].progress = 100; } 
        else { orders[currentOrderIdx].progress = 0; }
      }
      isMachineWorking = false; lastReadyTime = millis(); 
    }
    else if (d.startsWith("<STOCK:")) {
      int s1=d.indexOf('|'), s2=d.indexOf('|',s1+1), s3=d.indexOf('|',s2+1), s4=d.indexOf('|',s3+1), s5=d.indexOf('|',s4+1), s6=d.indexOf('|',s5+1), s7=d.indexOf('|',s6+1), s8=d.indexOf('|',s7+1);
      mSt=d.substring(7,s1).toInt(); 
      bS_L=d.substring(s1+1,s2).toInt(); 
      chS=d.substring(s2+1,s3).toInt(); 
      oS=d.substring(s3+1,s4).toInt(); 
      lS=d.substring(s4+1,s5).toInt(); 
      tS=d.substring(s5+1,s6).toInt(); 
      bS_H=d.substring(s6+1,s7).toInt();
      kSt=d.substring(s7+1,s8).toInt(); 
      mSt_sauce=d.substring(s8+1,d.indexOf(">")).toInt(); 
    }
    else if (d.startsWith("<ALERT:")) { systemAlert = d.substring(7, d.indexOf(">")); }
    else if (d.startsWith("<CLEARALERT>")) { systemAlert = ""; }
  }
}
