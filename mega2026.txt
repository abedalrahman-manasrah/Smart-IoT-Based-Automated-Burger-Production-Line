// دالة لتحديث الشاشة ونسبة إنجاز الطلب وإرسالها للتطبيق

  lcd.clear(); lcd.setCursor(3, 0); lcd.print("Burger Line"); 
  lcd.setCursor(0, 2); lcd.print(msg); 
  lcd.setCursor(0, 3); lcd.print("[");

// دالة لتحريك السيرفو بنعومة لتجنب الاهتزاز المفاجئ
void moveServoSmooth(Servo &myServo, int startAngle, int endAngle) {
  if (startAngle < endAngle) { for (int pos = startAngle; pos <= endAngle; pos += SERVO_STEP) { myServo.write(pos); delay(SERVO_SPEED); } } 
  else { for (int pos = startAngle; pos >= endAngle; pos -= SERVO_STEP) { myServo.write(pos); delay(SERVO_SPEED); } }
  myServo.write(endAngle); 
}

// دالة لتحريك السيرفو بسرعة مخصصة
void moveServoSpeed(Servo &myServo, int startAngle, int endAngle, int stepSize, int spd) {
  if (startAngle < endAngle) { for (int pos = startAngle; pos <= endAngle; pos += stepSize) { myServo.write(pos); delay(spd); } } 
  else { for (int pos = startAngle; pos >= endAngle; pos -= stepSize) { myServo.write(pos); delay(spd); } }
  myServo.write(endAngle);
}

// دالة لعمل اهتزاز (رج) للسيرفو لفك انحشار المواد
void shakeServo(Servo &myServo, int baseAngle) {
  for (int i = 0; i < SHAKE_TIMES; i++) { 
      myServo.write(baseAngle - SHAKE_ANGLE); delay(SHAKE_SPEED); 
      myServo.write(baseAngle + SHAKE_ANGLE); delay(SHAKE_SPEED); 
  }
  myServo.write(baseAngle); 
}

// 📡 8. الفلتر الاحترافي لقراءة الألتراسونيك (Trimmed Mean Filter)
// ============================================================================
long readUltrasonic(int trig, int echo) {
  const int NUM_READS = 5;
  long readings[NUM_READS];
  int validCount = 0;

  for (int i = 0; i < NUM_READS; i++) {
    digitalWrite(trig, LOW); delayMicroseconds(2);
    digitalWrite(trig, HIGH); delayMicroseconds(10);
    digitalWrite(trig, LOW);
    long duration = pulseIn(echo, HIGH, 30000); 
    
    if (duration > 100) { 
      readings[validCount] = duration;
      validCount++;
    }
    delay(15); 
  }

  if (validCount == 0) return 999; 
  if (validCount == 1) return (readings[0] * 0.034 / 2); 

  for (int i = 0; i < validCount - 1; i++) {
    for (int j = 0; j < validCount - i - 1; j++) {
      if (readings[j] > readings[j + 1]) {
        long temp = readings[j];
        readings[j] = readings[j + 1];
        readings[j + 1] = temp;
      }
    }
  }

  long finalDuration = 0;
  
  if (validCount >= 3) {
    long sum = 0;
    for (int i = 1; i < validCount - 1; i++) {
      sum += readings[i];
    }
    finalDuration = sum / (validCount - 2);
  } else { 
    finalDuration = (readings[0] + readings[1]) / 2;
  }

  return (finalDuration * 0.034 / 2); 
}

// 🔍 6. الفحص الذاتي (Hardware Test) عند الإقلاع
// ============================================================================
void hardwareTest() {
  lcd.clear(); lcd.setCursor(1, 0); lcd.print("SYSTEM DIAGNOSTIC");
  
  setRGB(0, 255, 255); lcd.setCursor(0, 1); lcd.print("1. Lasers: Testing..");
  enableLasers(); delay(800);     

  int allLdrs[] = {ldrStart, ldrBunL, ldrSauce, ldrMeat, ldrCheese, ldrOnion, ldrLettuce, ldrTomato, ldrBunH, ldrEnd};
  bool hadError = false; 

  while(true) {
    bool currentError = false;
    int errorPin = -1;

    for(int i = 0; i < 10; i++) {
      if(digitalRead(allLdrs[i]) == LASER_BLOCKED) { 
        currentError = true; 
        errorPin = i; 
        break; 
      }
    }

    if(currentError) { 
      hadError = true;
      setRGB(255, 0, 0); 
      lcd.clear();
      lcd.setCursor(0, 0); lcd.print("CRITICAL ERROR!");
      lcd.setCursor(0, 1); lcd.print("Laser Blocked/Broken");
      lcd.setCursor(0, 2); lcd.print("Check Sensor A" + String(errorPin)); 
      lcd.setCursor(0, 3); lcd.print("Waiting for fix...");
      beep(300); delay(300); 
    } else { 
      if(hadError) {
        setRGB(0, 255, 0);
        lcd.clear();
        lcd.setCursor(3, 1); lcd.print("SENSOR FIXED!");
        lcd.setCursor(2, 2); lcd.print("Resuming Boot...");
        beep(100); delay(50); beep(100); delay(50); beep(400); delay(1500); 
        lcd.clear(); lcd.setCursor(1, 0); lcd.print("SYSTEM DIAGNOSTIC");
        setRGB(0, 255, 255); lcd.setCursor(0, 1); lcd.print("1. Lasers: ");
      }
      lcd.print("OK"); beep(200); 
      break; 
    }
  }
  disableLasers(); 

  setRGB(0, 0, 255); lcd.setCursor(0, 2); lcd.print("2. Conveyor: Testing");
  digitalWrite(conveyorIN1, HIGH); digitalWrite(conveyorIN2, LOW); analogWrite(conveyorENA, CONVEYOR_SPEED);
  delay(800); digitalWrite(conveyorIN1, LOW); digitalWrite(conveyorIN2, LOW); analogWrite(conveyorENA, 0); 
  lcd.setCursor(13, 2); lcd.print("OK     "); delay(200);

  setRGB(255, 255, 0); lcd.setCursor(0, 3); lcd.print("3. Servos: Testing  ");
  shakeServo(meatDisk, MEAT_DISK_HOME);
  
  botBunLower.write(15); botBunUpper.write(15); delay(200); topBunLower.write(15); topBunUpper.write(15); delay(200);
  meatDisk.write(15); meatDrop.write(15); delay(400);
  
  botBunLower.write(BOT_BUN_LOWER_CLOSE); botBunUpper.write(BOT_BUN_UPPER_PRESS); delay(200); 
  topBunLower.write(TOP_BUN_LOWER_CLOSE); topBunUpper.write(TOP_BUN_UPPER_PRESS); delay(200);
  
  botBunLower.write(BOT_BUN_LOWER_CLOSE); botBunUpper.write(BOT_BUN_UPPER_RELEASE); delay(200); 
  topBunLower.write(TOP_BUN_LOWER_CLOSE); topBunUpper.write(TOP_BUN_UPPER_RELEASE); delay(200);
  
  meatDisk.write(MEAT_DISK_HOME); meatDrop.write(MEAT_DROP_CLOSE); delay(400);
  lcd.setCursor(11, 3); lcd.print("OK       "); delay(500);

  lcd.clear(); lcd.setCursor(1, 0); lcd.print("SYSTEM DIAGNOSTIC"); setRGB(255, 0, 255); lcd.setCursor(0, 1); lcd.print("4. Steppers: Testing");
  for(int i = 0; i < 6; i++) {
    int delayUS = (i >= 4) ? STEP_DELAY_SAUCE_US : STEP_DELAY_SALAD_US;
    int testSteps = (i >= 4) ? 100 : 10; 
    
    digitalWrite(dirPins[i], HIGH); 
    for(int s=0; s < testSteps; s++){ digitalWrite(stepPins[i], HIGH); delayMicroseconds(delayUS); digitalWrite(stepPins[i], LOW); delayMicroseconds(delayUS); }
    delay(50); digitalWrite(dirPins[i], LOW); 
    for(int s=0; s < testSteps; s++){ digitalWrite(stepPins[i], HIGH); delayMicroseconds(delayUS); digitalWrite(stepPins[i], LOW); delayMicroseconds(delayUS); }
    delay(100);
  }
  lcd.setCursor(13, 1); lcd.print("OK     ");
  
  lcd.clear(); lcd.setCursor(3, 1); lcd.print("ALL SYSTEMS GO!"); 
  setRGB(0, 255, 0); beep(200); delay(100); beep(200); delay(100); beep(600); delay(1500);
}

  delay(800); 
}

void dropAirlockBun(Servo &lower, Servo &upper, int pressAngle, int releaseAngle, int openAngle, int closeAngle, int ultraIdx) { 
  bool dropped = false;
  int currentEmptyLimit = (ultraIdx == 1) ? EMPTY_DIST_BOT_BUN_CM : EMPTY_DIST_TOP_BUN_CM;
  String bunName = (ultraIdx == 1) ? "BOT BUN" : "TOP BUN";

  while (!dropped) {
    int attempts = 0;
    
    while (attempts < 3) {
      long d1 = readUltrasonic(trigPins[ultraIdx], echoPins[ultraIdx]); delay(15);
      long d2 = readUltrasonic(trigPins[ultraIdx], echoPins[ultraIdx]); delay(15);
      long d3 = readUltrasonic(trigPins[ultraIdx], echoPins[ultraIdx]);
      long distBefore = (d1 + d2 + d3) / 3;

      moveServoSpeed(upper, releaseAngle, pressAngle, BUN_NORMAL_STEP, BUN_NORMAL_SPEED); 
      delay(500); 
      
      lower.write(openAngle); 
      delay(WAIT_DROP); 
      
      moveServoSpeed(lower, openAngle, closeAngle, BUN_NORMAL_STEP, BUN_NORMAL_SPEED);
      
      moveServoSpeed(upper, pressAngle, releaseAngle, BUN_NORMAL_STEP, BUN_NORMAL_SPEED);
      delay(WAIT_BREAD_FALL);

      delay(600); 
      
      long a1 = readUltrasonic(trigPins[ultraIdx], echoPins[ultraIdx]); delay(15);
      long a2 = readUltrasonic(trigPins[ultraIdx], echoPins[ultraIdx]); delay(15);
      long a3 = readUltrasonic(trigPins[ultraIdx], echoPins[ultraIdx]);
      long distAfter = (a1 + a2 + a3) / 3;

      if (distAfter >= distBefore + 1 || distAfter >= currentEmptyLimit) {
        dropped = true;
        break; 
      } else {
        attempts++; 
        beep(100); delay(100); beep(100); 
      }
    }
    
    if (!dropped) {
      setRGB(255, 0, 0); 
      lcd.clear(); 
      lcd.setCursor(0, 0); lcd.print("ERROR: BUN STUCK!");
      lcd.setCursor(0, 1); lcd.print("Check " + bunName);
      lcd.setCursor(0, 3); lcd.print("Fix & Press YES");
      
      while(digitalRead(btnYes) == HIGH) { beep(200); delay(300); }
      
      setRGB(0, 255, 0); 
      lcd.clear(); 
      updateUI("Retrying Bun..", (ultraIdx == 1) ? 15 : 80);
    }
  }
}

void shakeConveyorTray() {
  for (int i = 0; i < 4; i++) {
    digitalWrite(conveyorIN1, HIGH); digitalWrite(conveyorIN2, LOW); analogWrite(conveyorENA, 255); delay(130); 
    digitalWrite(conveyorIN1, LOW); digitalWrite(conveyorIN2, HIGH); analogWrite(conveyorENA, 255); delay(130); 
  }
  digitalWrite(conveyorIN1, LOW); digitalWrite(conveyorIN2, LOW); analogWrite(conveyorENA, 0); delay(400); 
}

void moveToStation(int ldrPin, String stationName) {
  digitalWrite(conveyorIN1, HIGH); digitalWrite(conveyorIN2, LOW); analogWrite(conveyorENA, CONVEYOR_SPEED);
  unsigned long moveStart = millis();

  int trackingLasers[] = {A1, A2, A3, A4, A5, A6, A7, A8};

  while(digitalRead(ldrPin) == LASER_REACHING) {
    for(int i=0; i<8; i++) {
       if(digitalRead(trackingLasers[i]) == LASER_BLOCKED) { moveStart = millis(); }
    }

    if(millis() - moveStart > ANTI_JAM_TIMEOUT) {
      digitalWrite(conveyorIN1, LOW); analogWrite(conveyorENA, 0); 
      setRGB(255, 0, 0);
      lcd.clear();
      lcd.setCursor(2, 0); lcd.print("SYSTEM HALTED!");
      lcd.setCursor(0, 1); lcd.print("TRAY JAM DETECTED");
      lcd.setCursor(0, 2); lcd.print("At: " + stationName);
      lcd.setCursor(0, 3); lcd.print("Fix & Press RESET");
      Serial1.println("<ALERT:توقف طارئ! الصينية عالقة. يرجى حل المشكلة والضغط على Reset>");
      Serial1.flush();
      while(true) { beep(300); delay(300); } 
    }
  }
  digitalWrite(conveyorIN1, LOW); analogWrite(conveyorENA, 0); delay(300); 
}

void dispenseSauceWithZigZag(int idx, int totalSteps) {
  int passes = 4; 
  int initialSteps = 200; 
  int stepsPerPass = (totalSteps - initialSteps) / passes; 

  if (idx == 4 || idx == 5) { digitalWrite(dirPins[idx], LOW); } 
  else { digitalWrite(dirPins[idx], HIGH); }

  for(int i = 0; i < initialSteps; i++){ 
    digitalWrite(stepPins[idx], HIGH); delayMicroseconds(STEP_DELAY_SAUCE_US); 
    digitalWrite(stepPins[idx], LOW); delayMicroseconds(STEP_DELAY_SAUCE_US); 
  }
  delay(2500); 

  for (int p = 0; p < passes; p++) {
    if (p % 2 == 0) {
      digitalWrite(conveyorIN1, HIGH); digitalWrite(conveyorIN2, LOW); 
    } else {
      digitalWrite(conveyorIN1, LOW); digitalWrite(conveyorIN2, HIGH); 
    }
    analogWrite(conveyorENA, 130); 

    for(int i = 0; i < stepsPerPass; i++){ 
      digitalWrite(stepPins[idx], HIGH); delayMicroseconds(STEP_DELAY_SAUCE_US); 
      digitalWrite(stepPins[idx], LOW); delayMicroseconds(STEP_DELAY_SAUCE_US); 
    }

    delay(350); 
    digitalWrite(conveyorIN1, LOW); digitalWrite(conveyorIN2, LOW); analogWrite(conveyorENA, 0);
    delay(1000);
  }

  delay(3000); 
  
  if (idx == 4 || idx == 5) { digitalWrite(dirPins[idx], HIGH); } 
  else { digitalWrite(dirPins[idx], LOW); }
  for(int i = 0; i < 400; i++){ 
    digitalWrite(stepPins[idx], HIGH); delayMicroseconds(300); 
    digitalWrite(stepPins[idx], LOW); delayMicroseconds(300); 
  }

  if(idx == 4) { currentPosK += totalSteps; EEPROM.put(EEPROM_K, currentPosK); } 
  if(idx == 5) { currentPosM += totalSteps; EEPROM.put(EEPROM_M, currentPosM); }
}

void runStepperReal(int idx, int steps) {
  if (idx == 4 || idx == 5) { digitalWrite(dirPins[idx], LOW); } 
  else { digitalWrite(dirPins[idx], HIGH); } 
  
  int delayUS = (idx >= 4) ? STEP_DELAY_SAUCE_US : STEP_DELAY_SALAD_US;
  
  for(int i = 0; i < steps; i++){ digitalWrite(stepPins[idx], HIGH); delayMicroseconds(delayUS); digitalWrite(stepPins[idx], LOW); delayMicroseconds(delayUS); }
  
  if(idx == 4) { currentPosK += steps; EEPROM.put(EEPROM_K, currentPosK); } 
  if(idx == 5) { currentPosM += steps; EEPROM.put(EEPROM_M, currentPosM); }
}

void parseOrder(String cmd) {
  int nameStart = cmd.indexOf("NAME:") + 5; int nameEnd = cmd.indexOf("|BITS:");
  currentName = cmd.substring(nameStart, nameEnd); 
  currentBits = cmd.substring(cmd.indexOf("BITS:") + 5, cmd.indexOf(">")); 
}

bool checkStockBeforeOrder() {
  setRGB(0, 0, 255); lcd.clear(); lcd.setCursor(0, 1); lcd.print("Checking Stock...");
  
  String missingItem = "";

  if (readUltrasonic(trigPins[0], echoPins[0]) >= EMPTY_DIST_MEAT_CM) missingItem = "MEAT";
  else if (readUltrasonic(trigPins[1], echoPins[1]) >= EMPTY_DIST_BOT_BUN_CM) missingItem = "BOT BUN";
  else if (readUltrasonic(trigPins[6], echoPins[6]) >= EMPTY_DIST_TOP_BUN_CM) missingItem = "TOP BUN";
  
  else if (currentBits.length() >= 6) {
    if (currentBits.charAt(0) == '1' && readUltrasonic(trigPins[2], echoPins[2]) >= EMPTY_DIST_SALAD_CM) missingItem = "CHEESE";
    else if (currentBits.charAt(1) == '1' && readUltrasonic(trigPins[3], echoPins[3]) >= EMPTY_DIST_SALAD_CM) missingItem = "ONION";
    else if (currentBits.charAt(2) == '1' && readUltrasonic(trigPins[4], echoPins[4]) >= EMPTY_DIST_SALAD_CM) missingItem = "LETTUCE";
    else if (currentBits.charAt(3) == '1' && readUltrasonic(trigPins[5], echoPins[5]) >= EMPTY_DIST_SALAD_CM) missingItem = "TOMATO";
    
    else if (currentBits.charAt(4) == '1' && currentPosK + stepQuantities[4] > MAX_SYRINGE_STEPS) missingItem = "KETCHUP";
    else if (currentBits.charAt(5) == '1' && currentPosM + stepQuantities[5] > MAX_SYRINGE_STEPS) missingItem = "MAYO";
  }

  if (missingItem != "") {
    setRGB(255, 0, 0); 
    lcd.clear(); 
    lcd.setCursor(2, 0); lcd.print("! STOCK EMPTY !"); 
    lcd.setCursor(0, 2); lcd.print("Missing: " + missingItem);
    lcd.setCursor(0, 3); lcd.print("Refill & Try Again");
    beep(1000); delay(2000); 
    return false; 
  }

  lcd.clear(); lcd.setCursor(3, 1); lcd.print("Stock OK"); delay(500); return true; 
}

void updateStockToESP() {
  String stockPayload = "<STOCK:";
  for(int i = 0; i < 7; i++) {
    long dist = readUltrasonic(trigPins[i], echoPins[i]);
    
    int emptyLim, fullLim;
    
    if (i == 0) { emptyLim = EMPTY_DIST_MEAT_CM; fullLim = FULL_DIST_MEAT_CM; }
    else if (i == 1) { emptyLim = EMPTY_DIST_BOT_BUN_CM; fullLim = FULL_DIST_BOT_BUN_CM; }
    else if (i == 6) { emptyLim = EMPTY_DIST_TOP_BUN_CM; fullLim = FULL_DIST_TOP_BUN_CM; }
    else { emptyLim = EMPTY_DIST_SALAD_CM; fullLim = FULL_DIST_SALAD_CM; }

    int pct = map(dist, emptyLim, fullLim, 0, 100); pct = constrain(pct, 0, 100);
    if (dist >= emptyLim || dist == 999) pct = 0; 
    stockPayload += String(pct) + "|";
  }
  
  float usableSteps = MAX_SYRINGE_STEPS - SAUCE_OFFSET_STEPS;
  int kPer = constrain(100 - (((float)currentPosK - SAUCE_OFFSET_STEPS) / usableSteps * 100), 0, 100);
  int mPer = constrain(100 - (((float)currentPosM - SAUCE_OFFSET_STEPS) / usableSteps * 100), 0, 100);
  
  stockPayload += String(kPer) + "|" + String(mPer) + ">"; 
  Serial1.println(stockPayload); Serial1.flush(); 
}

  preFetchMeat(); 
  dropMeatReal(); 

  updateUI("Adding Veggies..", 60);
  if(currentBits.charAt(0) == '1') { moveToStation(ldrCheese, "Cheese Station"); runStepperReal(0, stepQuantities[0]); delay(2500); }
  if(currentBits.charAt(1) == '1') { moveToStation(ldrOnion, "Onion Station"); runStepperReal(1, stepQuantities[1]); delay(2500); }
  if(currentBits.charAt(2) == '1') { moveToStation(ldrLettuce, "Lettuce Station"); runStepperReal(2, stepQuantities[2]); delay(2500); }
  if(currentBits.charAt(3) == '1') { moveToStation(ldrTomato, "Tomato Station"); runStepperReal(3, stepQuantities[3]); delay(2500); }

  moveToStation(ldrBunH, "Top Bun Station");

  moveToStation(ldrEnd, "Delivery Area");

  beep(700); lcd.clear(); lcd.setCursor(4, 0); lcd.print("ORDER READY!"); 
  lcd.setCursor(0, 2); lcd.print("For: " + currentName); lcd.setCursor(0, 3); lcd.print(" Please Take Meal ");
  
  while(digitalRead(ldrEnd) == LASER_BLOCKED); 
  
  Serial1.println("<CLEARALERT>"); 
  Serial1.flush();
}

void handleLocalOrder() {
  Serial1.println("<BUSY>"); clearSerialBuffer(); setRGB(200, 80, 0); 
  
  currentBits = "000000"; 
  if(!checkStockBeforeOrder()) { Serial1.println("<READY>"); showWelcome(); return; } 
  
  isLocalOrder = true; unsigned long lastInteraction = millis();
  int quantity = 1; currentBits = "";
  lcd.clear(); lcd.setCursor(0, 0); lcd.print("Number of Meals?");
  
  while(true) {
    lcd.setCursor(0, 1); lcd.print("Qty: " + String(quantity) + "   ");
    if (digitalRead(btnUp) == LOW) { if(quantity < 2) quantity++; beep(100); lastInteraction = millis(); delay(250); }
    if (digitalRead(btnDown) == LOW) { if(quantity > 1) quantity--; beep(100); lastInteraction = millis(); delay(250); }
    if (digitalRead(btnYes) == LOW) { beep(200); Serial1.print("<KEYPAD_START:"); Serial1.print(quantity); Serial1.println(">"); break; } 
    if (digitalRead(btnNo) == LOW || (millis() - lastInteraction > TIMEOUT_MS)) { cancelTimeout(); return; } 
  }
  
  String menu[] = {"Cheese?", "Onion?", "Lettuce?", "Tomato?", "Ketchup?", "Mayo?"};
  for(int i=0; i<6; i++) {
    lcd.clear(); lcd.setCursor(0, 0); lcd.print("Add " + menu[i]);
    while(true) {
      if (digitalRead(btnYes) == LOW) { currentBits += "1"; beep(100); delay(600); lastInteraction=millis(); break; } 
      if (digitalRead(btnNo) == LOW)  { currentBits += "0"; beep(100); delay(600); lastInteraction=millis(); break; } 
      if (millis() - lastInteraction > TIMEOUT_MS) { cancelTimeout(); return; } 
    }
  }
  
  // 🌟 هنا تم التعديل: فحص المخزون قبل كل وجبة 🌟
  enableLasers();
  for(int q = 1; q <= quantity; q++) { 
    currentName = "Local " + String(q) + "/" + String(quantity);

    if(!checkStockBeforeOrder()) { 
       disableLasers();
       cancelTimeout(); 
       return; 
    }
    
    processOrderReal(); 
    
    clearSerialBuffer(); 
    Serial1.println("<READY>"); 
    Serial1.flush();
  }
  disableLasers(); 
  
  showWelcome();
}

// 🔄 12. حلقة التشغيل المستمرة (Main Loop)
// ============================================================================
void loop() {
  if (Serial1.available()) {
    String cmd = Serial1.readStringUntil('\n'); cmd.trim(); 
    if (cmd.length() > 0) {
      
      if (cmd.startsWith("<ORDER:")) {
        Serial1.println("<BUSY>"); 
        isLocalOrder = false; parseOrder(cmd); 
        
        if(checkStockBeforeOrder()) { 
          enableLasers(); 
          processOrderReal(); 
          disableLasers(); 
          clearSerialBuffer(); Serial1.println("<READY>"); Serial1.flush(); showWelcome(); 
        } 
        else { Serial1.println("<READY>"); Serial1.flush(); showWelcome(); }
      }
      
      else if (cmd.startsWith("<REFILL:")) {
        Serial1.println("<BUSY>");
        char sauce = cmd.charAt(8); 
        int stepperIdx = (sauce == 'K') ? 4 : 5;
        int limitPin = (sauce == 'K') ? limitKetchup : limitMayo;
        long *currentPos = (sauce == 'K') ? &currentPosK : &currentPosM;
        int eepromAddr = (sauce == 'K') ? EEPROM_K : EEPROM_M;
        
        if (cmd.indexOf(":UP>") != -1) {
          if(sauce == 'K') isKetchupUp = true;
          if(sauce == 'M') isMayoUp = true;
          lcd.clear(); setRGB(255, 150, 0); lcd.setCursor(0, 0); lcd.print(">> MAINTAINANCE <<"); lcd.setCursor(0, 2); lcd.print("Homing "); lcd.print(sauce);
          
          digitalWrite(dirPins[stepperIdx], HIGH); 
          
          long timeoutSteps = MAX_SYRINGE_STEPS + 5000; 
          while(digitalRead(limitPin) == HIGH && timeoutSteps > 0) { 
            digitalWrite(stepPins[stepperIdx], HIGH); delayMicroseconds(STEP_DELAY_SAUCE_US); 
            digitalWrite(stepPins[stepperIdx], LOW); delayMicroseconds(STEP_DELAY_SAUCE_US); 
            timeoutSteps--; 
          }
          *currentPos = 0; EEPROM.put(eepromAddr, *currentPos); 
          lcd.clear(); lcd.print("Reached TOP Limit!"); beep(500); delay(1000);
        }
        else if (cmd.indexOf(":SET:") != -1) {
          if(sauce == 'K') isKetchupUp = false;
          if(sauce == 'M') isMayoUp = false;
          lcd.clear(); setRGB(255, 150, 0); lcd.setCursor(0, 0); lcd.print(">> MAINTAINANCE <<");
          int val = cmd.substring(cmd.lastIndexOf(':') + 1, cmd.indexOf('>')).toInt(); 
          lcd.setCursor(0, 2); lcd.print("Set "); lcd.print(sauce); lcd.print(" to "); lcd.print(val); lcd.print("%");
          
          float usableSteps = MAX_SYRINGE_STEPS - SAUCE_OFFSET_STEPS;
          long targetPos = SAUCE_OFFSET_STEPS + (usableSteps * (100 - val) / 100); 
          if (val == 0) targetPos = MAX_SYRINGE_STEPS; 
          
          if (targetPos > MAX_SYRINGE_STEPS) targetPos = MAX_SYRINGE_STEPS;
          if (targetPos < 0) targetPos = 0;
          
          long stepsToMove = targetPos - *currentPos;
          
          if (stepsToMove > 0) { 
            digitalWrite(dirPins[stepperIdx], LOW);
            for(long i = 0; i < stepsToMove; i++) { digitalWrite(stepPins[stepperIdx], HIGH); delayMicroseconds(STEP_DELAY_SAUCE_US); digitalWrite(stepPins[stepperIdx], LOW); delayMicroseconds(STEP_DELAY_SAUCE_US); } 
          } 
          else if (stepsToMove < 0) { 
            digitalWrite(dirPins[stepperIdx], HIGH);
            stepsToMove = abs(stepsToMove); 
            for(long i = 0; i < stepsToMove; i++) { digitalWrite(stepPins[stepperIdx], HIGH); delayMicroseconds(STEP_DELAY_SAUCE_US); digitalWrite(stepPins[stepperIdx], LOW); delayMicroseconds(STEP_DELAY_SAUCE_US); } 
          }
          *currentPos = targetPos; EEPROM.put(eepromAddr, *currentPos); 
          lcd.clear(); lcd.print("Ready at "); lcd.print(val); lcd.print("%"); beep(200); delay(100); beep(200); delay(1000);
        }
        
        if (isKetchupUp || isMayoUp) { lcd.clear(); setRGB(255, 0, 0); lcd.setCursor(2, 1); lcd.print("REFILL ACTIVE!"); lcd.setCursor(0, 2); lcd.print("Set % from Web App"); } 
        else { clearSerialBuffer(); Serial1.println("<READY>"); Serial1.flush(); showWelcome(); }
      }
      cmd = "";
    }
  }

  if (digitalRead(btnYes) == LOW && !Serial1.available()) { 
    delay(200); beep(100);
    if (isKetchupUp || isMayoUp) {
      lcd.clear(); setRGB(255, 0, 0); lcd.setCursor(2, 1); lcd.print("SYSTEM LOCKED!"); lcd.setCursor(2, 2);
      lcd.print("Refill Active"); beep(500); delay(1500);
      lcd.clear(); setRGB(255, 0, 0); lcd.setCursor(2, 1); lcd.print("REFILL ACTIVE!"); lcd.setCursor(0, 2); lcd.print("Set % from Web App");
    } else { 
      handleLocalOrder();
    }
  }

  if (millis() - lastStockTime > 1500) { updateStockToESP(); lastStockTime = millis(); }
}
